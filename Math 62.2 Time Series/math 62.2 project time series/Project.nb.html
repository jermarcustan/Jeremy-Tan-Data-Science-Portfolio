Preprocessing the Data

# Read the CSV data
BDO = read.csv("BDO.csv")
ALI = read.csv("ALI.csv")

#Assign Date as the index of the dataframe
rownames(BDO) = BDO$Date
rownames(ALI) = ALI$Date

BDO = BDO[rev(rownames(BDO)), ]
ALI = ALI[rev(rownames(ALI)), ]
ALI
